			fragment 

			
#extension GL_OES_standard_derivatives : enable
precision mediump float;

uniform sampler2D TEXTURE_1;
varying vec4 color;
varying vec2 uv1;

void main(){
	vec4 diffuse = color;
	diffuse *= texture2D(TEXTURE_1, uv1);
	gl_FragColor = diffuse;
}
